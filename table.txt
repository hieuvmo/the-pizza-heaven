1. table category (
    category_id: int (primary key auto_increment),
    category_name: string,
)

2. table food (
	food_id:  int (primary key auto_increment),
	category_id: int (references category (id)),
	name: string
    thumbnail: string
	description: string 
	price: Array[string]
	size: Array[string]
	is_stock: boolean
)

3 table rating(
	rating_id: int
	user_id: int (phải mua mới được đánh giá)
	food_id: int
	star: int
	review: string
	is_anonymous: boolean
)

4. table user (
	user_id: int (nếu không tạo tài khoản thì để user_id âm)
	username: string  
	password: string
	firstName: string
    lastName: string
	phone: string 
	dob: string
	address: {
        noHouse: string, 
        ward: string, 
        district: string, 
        city: string
    } 
	gender: string;	
)

5. table order (
	order_id: int
	user_id: int
	totalPrice: string 	
)

6. table order_detail (
	oder_detail_id: int
	order_id: int
	food_id: int
	quantity: int
	note: string
	status: new(add to cart) | confirm | success | cancel
)
